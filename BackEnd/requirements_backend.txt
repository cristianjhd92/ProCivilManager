# ===============================================

# --- Dependencias ---
express                 # Servidor HTTP + enrutamiento (API REST)
cors                    # Habilita CORS para que el FrontEnd consuma la API
mongoose                # ODM para MongoDB (modelos/esquemas)
dotenv                  # Carga variables de entorno desde .env
jsonwebtoken            # Emisión/verificación de tokens JWT (auth)
bcryptjs                # Hash/sal de contraseñas
nodemailer              # Envío de correos electrónicos
pdfkit                  # Generación de PDF en el servidor
socket.io               # WebSockets/tiempo real (notificaciones, etc.)

# --- Dependencias para desarrollo ---
nodemon                 # Recarga automática en desarrollo (watch)

# --- Comandos sugeridos ---
#   npm install express cors mongoose dotenv jsonwebtoken bcryptjs nodemailer pdfkit socket.io
#   npm install -D nodemon
