# ===============================================
# Procivil Manager – FrontEnd (React + CRA + Tailwind)
# Nota: npm/yarn/pnpm NO leen este archivo; sirve como documentación.
# Para instalar usa los comandos indicados al final.
# ===============================================

# --- Dependencias ---
react                   # Librería de UI
react-dom               # Render de React al DOM
react-router-dom        # Ruteo del FrontEnd (SPA)
react-scripts           # Tooling de Create React App (build/dev/test)
recharts                # Gráficas
lucide-react            # Iconos (basado en Lucide)
jspdf                   # Generación de PDF en el cliente
jspdf-autotable         # Tablas en PDF (extensión de jsPDF)

# --- Dependencias de desarrollo ---
tailwindcss               # Framework CSS utilitario
postcss                   # Transformaciones CSS (pipeline)
autoprefixer              # Prefijos CSS para navegadores
@testing-library/react    # Utilidades de test para componentes
@testing-library/jest-dom # Matchers extra para Jest

# --- Comandos sugeridos ---
#   npm install react react-dom react-router-dom react-scripts recharts lucide-react jspdf jspdf-autotable
#   npm install -D tailwindcss postcss autoprefixer @testing-library/react @testing-library/jest-dom
